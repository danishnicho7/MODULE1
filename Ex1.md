## Program:
```cpp
#include <iostream>
#include <cmath>
using namespace std;

class GPSeries {
private:
    float a, r;
    int n;

public:
    void getInput();
    friend float findSum(GPSeries obj);  
};

void GPSeries::getInput() {
    cin >> a >> r >> n;
}
float findSum(GPSeries obj) {
    float sum;
    if (obj.r == 1) {
        sum = obj.a * obj.n; 
    } else {
        sum = obj.a * (pow(obj.r, obj.n) - 1) / (obj.r - 1);
    }
    return sum;
}

int main() {
    GPSeries gp;
    gp.getInput();

    float sum = findSum(gp);  
    cout << "The sum of GP series is " << sum << endl;

    return 0;
}
```
